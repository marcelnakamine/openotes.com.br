<?php 
    //MILISEGUNDOS
    date('u'); // 0 até 999

    //SEGUNDOS
    date('s'); // 0 até 59

    //MINUTOS
    date('i'); // 0 até 59

    //HORAS
    date('g'); // 0 até 12
    date('G'); // 0 até 23
    date('h'); // 00 até 12
    date('H'); // 00 até 23

    //DIA
    date('d'); // 01
    date('j'); // 1

    //MÊS
    date('F'); // January
    date('m'); // 01
    date('M'); // Jan
    date('n'); // 1

    //ANO
    date('Y'); // 2012
    date('y'); // 12 
 
    //DATA COMPLETA
    date('d/m/Y'); // 01/01/2012

    //HORÁRIO COMPLETO
    date('H:i:s:u'); // 23:59:59:000 
?>
